// Add an expense
void addExpense(struct Expense expenses[], int *expenseCount) {
if (*expenseCount >= 100) {
printf("\t\t\t\t\t\tExpense limit reached. Cannot add more expenses.\n");
return;
}
struct Expense newExpense;
newExpense.id = *expenseCount + 1;
printf("\t\t\t\t\t\tEnter date (DD/MM/YYYY): ");
scanf("%s", newExpense.date);
printf("\t\t\t\t\t\tEnter description: ");
scanf(" %[^\n]", newExpense.description);
printf("\t\t\t\t\t\tEnter amount: ");
scanf("%f", &newExpense.amount);
Your_Account_Balance -= newExpense.amount;
expenses[*expenseCount] = newExpense;
(*expenseCount)++;
printf("\t\t\t\t\t\t \033[1;32m Expense added successfully.\033[0m\n");
}
// Display expenses
void displayExpenses(struct Expense expenses[], int expenseCount) {
if (expenseCount == 0) {
printf("\t\t\t\t\t\t \033[1;31mNo expenses recorded.\033[0m\n");
return;
}
printf("\t\t\t\t\t\tExpense ID | Date | Description | Amount\n");
int i;
printf("\t\t\t\t\t\t--------------------------------------------------------------\n");
for ( i = 0; i < expenseCount; i++) {
printf("\t\t\t\t\t\t%-10d | %-10s | %20s | %.2f\n", expenses[i].id,
expenses[i].date, expenses[i].description, expenses[i].amount);
}
printf("\t\t\t\t\t\t--------------------------------------------------------------\n");
printf("\t\t\t\t\t\t Petty Cash Balance: %.2f\n", Your_Account_Balance);
}
// Search expenses by date
void searchExpensesByDate(struct Expense expenses[], int expenseCount, char searchDate[20]) {
if (expenseCount == 0) {
printf("\t\t\t\t\t\t No expenses recorded.\n");
return;
}
float totalAmount = 0.0;
int found = 0;
printf("\t\t\t\t\t\tExpense ID | Date | Description | Amount\n");
printf("\t\t\t\t\t\t--------------------------------------------------\n");
int i;
for (i = 0; i < expenseCount; i++) {
if (strcmp(expenses[i].date, searchDate) == 0) {
printf("\t\t\t\t\t\t%10d | %10s | %20s | %.2f\n", expenses[i].id,
expenses[i].date,expenses[i].description, expenses[i].amount);
totalAmount += expenses[i].amount;
found = 1;
}
}
printf("\t\t\t\t\t--------------------------------------------------------------\n");
if (found) {
printf("\t\t\t\t\t\tTotal amount for %s: %.2f\n", searchDate, totalAmount);
} else {
printf("\t\t\t\t\t\tNo records found for %s.\n", searchDate);
}
}
// Update an expense
void updateExpense(struct Expense expenses[], int expenseCount) {
int idToUpdate;
printf("\t\t\t\t\t\tEnter the ID of the expense to update: ");
scanf("%d", &idToUpdate);
if (idToUpdate <= 0 || idToUpdate > expenseCount) {
printf("\t\t\t\t\t\t \033[1;31mInvalid expense ID.\033[0m\n");
return;
}
struct Expense *expenseToUpdate = &expenses[idToUpdate - 1];
printf("\t\t\t\t\t\tEnter new date (DD/MM/YYYY): ");
scanf("%s", expenseToUpdate->date);
printf("\t\t\t\t\t\tEnter new description: ");
scanf(" %[^\n]", expenseToUpdate->description);
printf("\t\t\t\t\t\tEnter new amount: ");
scanf("%f", &expenseToUpdate->amount);
printf("\t\t\t\t\t\t \033[1;32mExpense updated successfully.\033[0m\n");
}
// Delete an expense
void deleteExpense(struct Expense expenses[], int *expenseCount, int idToDelete) {
if (idToDelete <= 0 || idToDelete > *expenseCount) {
printf("\t\t\t\t\t\tInvalid expense ID.\n");
return;
}
int i;
for ( i = idToDelete - 1; i < *expenseCount - 1; i++) {
expenses[i] = expenses[i + 1];
}
(*expenseCount)--;
printf("\t\t\t\t\t\t \033[1;32mExpense deleted successfully.\033[0m\n\n");
}
